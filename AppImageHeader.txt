#!/bin/sh
clear
HOMEPATH="`pwd`"
EXTRACTFILE=`mktemp`
echo ${EXTRACTFILE}
DIRPATH=`mktemp -u -p ${HOMEPATH}`
echo ${DIRPATH}
# ${NUMBEROFLINES} replaced from buildit.sh
tail -n +${NUMBEROFLINES} "$0" > ${EXTRACTFILE}
unsquashfs -d ${DIRPATH} ${EXTRACTFILE}
# export LD_LIBRARY_PATH="${DIRPATH}/usr/lib/"
`${DIRPATH}/AppRun`
rm ${EXTRACTFILE}
rm -r ${DIRPATH}
exit 1
